## .editorconfig

> root = true
>
> [*] // 匹配所有文件
>
> charset = utf-8  // 文件编码
>
> indent_style = space   // 空格缩进, 可选"space  tab"
>
> indent_size = 4  // 缩进空格为4
>
> end_of_line = lf  //  结尾换行符，可选"lf"、"cr"、"crlf" 
>
> insert_final_newline = true //  文件末尾添加一个空行 
>
> trim_trailing_whitespace = true   //   不保留行末的空格
